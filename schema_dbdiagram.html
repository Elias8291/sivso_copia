¿Por qué 3 FKs denormalizadas en solicitudes_vestuario?
Sin ellas, para agrupar por dependencia/partida/tipo hay que hacer JOIN con empleados (→ dependencia_id) y productos (→ partida_id + tipo_id). Con las 3 FKs directas, el GROUP BY cae en un índice compuesto de la misma tabla — los JOINs restantes solo sirven para etiquetas (nombres). Cero pérdida de datos.
¿Por qué delegados con FK directo?
En la base original, cada fila de delegado tiene exactamente una delegacion. Un FK directo es más simple y rápido que un pivot N:M. Si en el futuro un delegado puede servir en varias delegaciones, se agrega el pivot después.

// ═══════════════════════════════════════════════════════════════
// SIVSO – Esquema normalizado  (base: copiasivso)
// ═══════════════════════════════════════════════════════════════
// Tablas legacy de origen:
//   concentrado  → empleados + solicitudes_vestuario + partidas
//                  + partidas_por_ejercicio + partidas_especificas
//   propuesta    → productos + producto_precios + tipos_partida_especifica
//   delegacion   → dependencia_delegacion + delegaciones (+ link empleados)
//   delegado     → delegados (FK directo a delegaciones)
//   dependences  → dependencias
// ═══════════════════════════════════════════════════════════════


// ─── GRUPO 1: ORGANIZACIÓN ──────────────────────────────────

Table dependencias {
  id        bigint       [pk, increment]
  codigo    varchar(12)  [note: 'dependences.key']
  ur        int          [note: 'concentrado.ur']
  nombre    varchar(255) [not null]
  ur_texto  varchar(120) [note: 'concentrado.ur_dependencia']
  created_at timestamp
  updated_at timestamp

  indexes {
    ur           [name: 'idx_dep_ur']
    codigo       [name: 'idx_dep_codigo']
    (ur, nombre) [unique, name: 'uq_dep_ur_nombre']
  }
}

Table delegaciones {
  id     bigint      [pk, increment]
  clave  varchar(60) [unique, not null, note: 'delegacion.delegacion ∪ delegado.delegacion']
  nombre varchar(255)
  created_at timestamp
  updated_at timestamp
}

// Pivot N:M – una dependencia puede aparecer en varias delegaciones
Table dependencia_delegacion {
  dependencia_id bigint [not null, ref: > dependencias.id]
  delegacion_id  bigint [not null, ref: > delegaciones.id]

  indexes {
    (dependencia_id, delegacion_id) [pk]
  }
}

// ★ FK directo (1 delegado → 1 delegación; en legacy cada fila tiene una sola)
Table delegados {
  id              bigint       [pk, increment]
  delegacion_id   bigint       [ref: > delegaciones.id,
                                note: 'delegado.delegacion → delegaciones.clave']
  nombre_completo varchar(255) [not null, note: 'delegado.nombre']
  created_at timestamp
  updated_at timestamp
}


// ─── GRUPO 2: CATÁLOGO VESTUARIO ────────────────────────────

Table tipos_partida_especifica {
  id     bigint   [pk, increment]
  codigo smallint [unique, not null,
                   note: 'propuesta.partida_especifica (244, 246…)']
  nombre varchar(120)
  created_at timestamp
  updated_at timestamp
}

Table partidas {
  id          bigint [pk, increment]
  no_partida  int    [unique, not null,
                      note: 'concentrado.no_partida = propuesta.partida']
  descripcion text   [note: 'propuesta.descripcion → concentrado.partida_descripcion (fallback)']
  created_at timestamp
  updated_at timestamp
}

// Datos de la partida que varían por año fiscal
Table partidas_por_ejercicio {
  id                   bigint       [pk, increment]
  partida_id           bigint       [not null, ref: > partidas.id]
  anio                 smallint     [not null]
  no_partida_snapshot  int          [note: 'snapshot de partidas.no_partida']
  clave_como_se_uso    varchar(512) [note: 'concentrado.clave_partida']
  clave_para_ejercicio varchar(512)
  clave_presupuestal   int          [note: 'concentrado.clave_presupuestal']
  created_at timestamp
  updated_at timestamp

  indexes {
    (partida_id, anio)          [unique]
    (anio, no_partida_snapshot) [name: 'idx_ppe_anio_snap']
  }
}

// Cada clave de ejercicio (concentrado.clave2025) con su descripción de línea
Table partidas_especificas {
  id            bigint       [pk, increment]
  partida_id    bigint       [not null, ref: > partidas.id]
  anio          smallint     [not null]
  clave         varchar(512) [not null, note: 'concentrado.clave2025']
  descripcion   text         [note: 'concentrado.descripcion (nivel línea)']
  clave_partida varchar(512) [note: 'concentrado.clave_partida de esa línea']
  created_at timestamp
  updated_at timestamp

  indexes {
    (partida_id, anio) [name: 'idx_pe_partida_anio']
    (anio, clave)      [name: 'idx_pe_anio_clave']
  }
}

Table productos {
  id                          bigint       [pk, increment]
  partida_id                  bigint       [not null, ref: > partidas.id]
  tipo_partida_especifica_id  bigint       [not null, ref: > tipos_partida_especifica.id]
  lote                        int
  descripcion                 text         [not null, note: 'propuesta.descripcion']
  marca                       varchar(120) [note: 'propuesta.marca']
  unidad_medida               varchar(30)  [note: 'propuesta.unidad']
  codigo                      varchar(60)  [note: 'propuesta.codigo']
  medida                      varchar(10)  [note: 'propuesta.medida']
  legacy_propuesta_id         bigint       [unique, note: 'propuesta.id – trazabilidad']
  created_at timestamp
  updated_at timestamp

  indexes {
    (partida_id, tipo_partida_especifica_id) [name: 'idx_prod_partida_tipo']
    codigo                                    [name: 'idx_prod_codigo']
  }
}

Table producto_precios {
  id              bigint        [pk, increment]
  producto_id     bigint        [not null, ref: > productos.id]
  anio            smallint      [not null]
  precio_unitario decimal(12,2) [not null, note: 'propuesta.precio_unitario']
  subtotal        decimal(14,2) [note: 'propuesta.subtotal']
  proveedor       varchar(120)  [note: 'propuesta.proveedor']
  created_at timestamp
  updated_at timestamp

  indexes {
    (producto_id, anio) [unique]
    anio                [name: 'idx_pp_anio']
  }
}


// ─── GRUPO 3: EMPLEADOS Y SOLICITUDES ───────────────────────

Table empleados {
  id               bigint       [pk, increment]
  nue              varchar(30)  [note: 'concentrado.nue / delegacion.nue']
  nombre           varchar(120) [note: 'concentrado.nombre_trab']
  apellido_paterno varchar(120) [note: 'concentrado.apellp_trab']
  apellido_materno varchar(120) [note: 'concentrado.apellm_trab']
  dependencia_id   bigint       [ref: > dependencias.id,
                                 note: 'via concentrado.dependencia → dependencias']
  delegacion_id    bigint       [ref: > delegaciones.id,
                                 note: 'via delegacion.nue → delegaciones']
  ur               int          [note: 'concentrado.ur – histórico']
  created_at timestamp
  updated_at timestamp

  indexes {
    nue            [name: 'idx_emp_nue']
    dependencia_id [name: 'idx_emp_dep']
    delegacion_id  [name: 'idx_emp_del']
    ur             [name: 'idx_emp_ur']
  }
}

Table cupos_presupuesto {
  id                          bigint        [pk, increment]
  dependencia_id              bigint        [not null, ref: > dependencias.id]
  partida_id                  bigint        [not null, ref: > partidas.id]
  tipo_partida_especifica_id  bigint        [not null, ref: > tipos_partida_especifica.id]
  anio                        smallint      [not null]
  cantidad_limite             decimal(15,2) [not null]
  created_at timestamp
  updated_at timestamp

  indexes {
    (dependencia_id, partida_id, tipo_partida_especifica_id, anio)
      [unique, name: 'uq_cupo_dep_part_tipo_anio']
  }
}

// ─── TABLA DE HECHOS ────────────────────────────────────────
// Cada fila = 1 ítem de vestuario asignado a un empleado
// Legacy: concentrado (una fila por empleado × producto × partida)
//
// ★ Las 3 FKs denormalizadas eliminan JOINs costosos
//   en los GROUP BY de reportes de partidas

Table solicitudes_vestuario {
  id bigint [pk, increment]

  // ── Relaciones principales
  empleado_id            bigint [not null, ref: > empleados.id]
  producto_id            bigint [not null, ref: > productos.id]
  partida_especifica_id  bigint [ref: > partidas_especificas.id,
                                 note: 'clave de ejercicio del ítem']

  // ── ★ FKs denormalizadas (GROUP BY rápido sin JOINs)
  dependencia_id              bigint [ref: > dependencias.id,
                                      note: '= empleado.dependencia_id al crear registro']
  partida_id                  bigint [ref: > partidas.id,
                                      note: '= producto.partida_id']
  tipo_partida_especifica_id  bigint [ref: > tipos_partida_especifica.id,
                                      note: '= producto.tipo_partida_especifica_id']

  // ── Datos de la solicitud
  anio             smallint
  talla            varchar(20)
  cantidad         int           [default: 1]
  precio_unitario  decimal(12,2) [note: 'concentrado.precio_unitario']
  importe          decimal(14,2) [note: 'concentrado.importe']
  iva              decimal(12,2) [note: 'concentrado.iva']
  importe_total    decimal(14,2) [note: 'concentrado.total']

  // ── Snapshots legado (trazabilidad)
  no_partida_snapshot            int          [note: 'concentrado.no_partida']
  clave_partida_snapshot         varchar(512) [note: 'concentrado.clave_partida']
  clave_para_ejercicio_snapshot  varchar(512) [note: 'concentrado.clave2025']

  // ── Trazabilidad
  legacy_concentrado_id bigint [unique, note: 'concentrado.id original']

  created_at timestamp
  updated_at timestamp

  indexes {
    (anio, dependencia_id, partida_id, tipo_partida_especifica_id)
      [name: 'sv_resumen_idx', note: 'GROUP BY sin JOIN en reportes']
    (empleado_id, anio) [name: 'idx_sv_emp_anio']
    (producto_id, anio) [name: 'idx_sv_prod_anio']
  }
}

Resumen de simplificaciones:
1. delegados → FK directo a delegaciones (sin pivot N:M, ya que en los datos originales cada delegado tiene una sola delegación)
2. solicitudes_vestuario → 3 FKs denormalizadas (dependencia_id, partida_id, tipo_partida_especifica_id) para que los reportes de partidas hagan GROUP BY sobre un índice compuesto de la misma tabla
3. partidas_especificas → FK directo a partidas (no pasa por partidas_por_ejercicio), más simple para consultas de línea
